var fs = require('fs');

module.exports = function(dir, filter, callback) {
  var regex = new RegExp("\\." + filter + "$");
  fs.readdir(dir, function(err, list) {
    if (err) {
      return callback(err, null); 
    }
    else {
      results = [];
      for (i = 0; i < list.length; i++) {
        if (regex.test(list[i])) {
          results.push(list[i]);
        }
      }
    callback(null, results);
  }
});
};
